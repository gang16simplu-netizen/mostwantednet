async function startRec() { const videoStream = document.getElementById('vid').srcObject; if (!videoStream) { alert("Camera nu este activă. Reîmprospătează pagina."); return; } // Detectăm tipul suportat let mimeType = MediaRecorder.isTypeSupported('video/webm;codecs=vp9') ? 'video/webm;codecs=vp9' : 'video/webm'; chunks = []; try { rec = new MediaRecorder(videoStream, { mimeType: mimeType }); } catch (e) { rec = new MediaRecorder(videoStream); // Fallback } rec.ondataavailable = (e) => { if (e.data.size > 0) chunks.push(e.data); }; rec.onstop = () => { document.getElementById('s2').classList.remove('active'); document.getElementById('s3').classList.add('active'); }; rec.start(); document.getElementById('btn2').innerText = "[ ÎNREGISTRARE ACTIVĂ... ]"; document.getElementById('btn2').disabled = true; // Prevenim dublu-click // Timer 15 secunde setTimeout(() => { if(rec && rec.state === "recording") { rec.stop(); } }, 15000); }
